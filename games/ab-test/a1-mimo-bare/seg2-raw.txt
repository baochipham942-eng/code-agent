class Player {
  constructor(x, y) {
    this.x = x;
    this.y = y;
    this.vx = 0;
    this.vy = 0;
    this.width = 20;
    this.height = 40;
    this.abilities = { doubleJump: false };
    this.jumpsRemaining = 1;
    this.grounded = false;
    this.coyoteTimer = 0;
    this.jumpBufferTimer = 0;
    this.alive = true;
    this.score = 0;
    this.lives = 3;
    this.animTime = 0;
  }

  update(dt, input, level) {
    if (!this.alive) return;
    const accel = 800;
    const friction = 0.85;
    const gravity = 2000;
    const coyoteTime = 0.1;
    const jumpBufferTime = 0.1;

    if (input.left) this.vx -= accel * dt;
    else if (input.right) this.vx += accel * dt;
    else this.vx *= friction;

    this.vy += gravity * dt;
    if (this.vy > 1200) this.vy = 1200;

    if (this.grounded) this.coyoteTimer = coyoteTime;
    else this.coyoteTimer -= dt;

    if (input.jumpPressed) this.jumpBufferTimer = jumpBufferTime;
    else this.jumpBufferTimer -= dt;

    if (this.jumpBufferTimer > 0 && this.coyoteTimer > 0) {
      this.jump();
      this.jumpBufferTimer = 0;
      this.coyoteTimer = 0;
    }

    this.x += this.vx * dt;
    this.y += this.vy * dt;
    this.grounded = false;

    for (const p of level.platforms) {
      if (
        this.x + this.width > p.x &&
        this.x < p.x + p.w &&
        this.y + this.height > p.y &&
        this.y < p.y + p.h
      ) {
        const overlapLeft = this.x + this.width - p.x;
        const overlapRight = p.x + p.w - this.x;
        const overlapTop = this.y + this.height - p.y;
        const overlapBottom = p.y + p.h - this.y;
        const minOverlap = Math.min(overlapLeft, overlapRight, overlapTop, overlapBottom);

        if (minOverlap === overlapTop && this.vy >= 0) {
          this.y = p.y - this.height;
          this.vy = 0;
          this.grounded = true;
          this.jumpsRemaining = this.abilities.doubleJump ? 2 : 1;
        } else if (minOverlap === overlapBottom && this.vy < 0) {
          this.y = p.y + p.h;
          this.vy = 0;
        } else if (minOverlap === overlapLeft) {
          this.x = p.x - this.width;
          this.vx = 0;
        } else if (minOverlap === overlapRight) {
          this.x = p.x + p.w;
          this.vx = 0;
        }
      }
    }

    this.animTime += dt;
    if (this.y > 2000) this.takeDamage();
  }

  jump() {
    if (this.jumpsRemaining > 0) {
      this.vy = -650;
      this.jumpsRemaining--;
      this.grounded = false;
    } else if (this.abilities.doubleJump && !this.grounded && this.jumpsRemaining === 0) {
      this.vy = -650;
      this.jumpsRemaining = -1;
    }
  }

  tryStompOrCollideEnemy(enemy) {
    if (!enemy.alive || !this.alive) return null;
    const fromAbove = this.vy > 0 && this.y + this.height - this.vy * 0.016 < enemy.y + enemy.height * 0.4;
    if (fromAbove) {
      enemy.alive = false;
      this.vy = -400;
      this.score += 100;
      return { type: 'stomp', x: enemy.x, y: enemy.y };
    } else if (
      this.x + this.width > enemy.x &&
      this.x < enemy.x + enemy.width &&
      this.y + this.height > enemy.y &&
      this.y < enemy.y + enemy.height
    ) {
      this.takeDamage();
      return { type: 'hit', x: this.x, y: this.y };
    }
    return null;
  }

  tryBumpBlock(block) {
    if (block.used) return null;
    if (
      this.vy < 0 &&
      this.x + this.width > block.x &&
      this.x < block.x + block.size &&
      this.y < block.y + block.size &&
      this.y > block.y
    ) {
      block.used = true;
      this.vy = 0;
      this.abilities.doubleJump = true;
      this.jumpsRemaining = 2;
      return { type: 'bump', x: block.x + block.size / 2, y: block.y };
    }
    return null;
  }

  takeDamage() {
    this.lives--;
    if (this.lives <= 0) {
      this.alive = false;
    } else {
      this.vy = -500;
    }
  }

  render(ctx) {
    const cx = this.x + this.width / 2;
    const headY = this.y + 6;
    const bodyTop = this.y + 12;
    const bodyBot = this.y + 26;
    const moving = Math.abs(this.vx) > 50;
    const jumping = this.vy < 0;
    const swing = moving ? Math.sin(this.animTime * 10) * 0.5 : 0;
    const legSwing = moving ? Math.sin(this.animTime * 10) * 8 : 0;

    ctx.strokeStyle = '#000';
    ctx.lineWidth = 2;
    ctx.lineCap = 'round';

    ctx.beginPath();
    ctx.arc(cx, headY, 6, 0, Math.PI * 2);
    ctx.stroke();

    ctx.beginPath();
    ctx.moveTo(cx, bodyTop);
    ctx.lineTo(cx, bodyBot);
    ctx.stroke();

    ctx.beginPath();
    ctx.moveTo(cx, bodyTop + 4);
    ctx.lineTo(cx - 10, bodyTop + 14 + (moving ? swing * 8 : 0));
    ctx.moveTo(cx, bodyTop + 4);
    ctx.lineTo(cx + 10, bodyTop + 14 - (moving ? swing * 8 : 0));
    ctx.stroke();

    if (jumping) {
      ctx.beginPath();
      ctx.moveTo(cx, bodyBot);
      ctx.lineTo(cx - 8, bodyBot + 12);
      ctx.moveTo(cx, bodyBot);
      ctx.lineTo(cx + 8, bodyBot + 12);
      ctx.stroke();
    } else {
      ctx.beginPath();
      ctx.moveTo(cx, bodyBot);
      ctx.lineTo(cx - 6 + legSwing, bodyBot + 14);
      ctx.moveTo(cx, bodyBot);
      ctx.lineTo(cx + 6 - legSwing, bodyBot + 14);
      ctx.stroke();
    }
  }
}

class Enemy {
  constructor(x, y, patrolMinX, patrolMaxX) {
    this.x = x;
    this.y = y;
    this.vx = 60;
    this.width = 24;
    this.height = 28;
    this.alive = true;
    this.patrolMinX = patrolMinX;
    this.patrolMaxX = patrolMaxX;
    this.animTime = 0;
  }

  update(dt) {
    if (!this.alive) return;
    this.x += this.vx * dt;
    if (this.x <= this.patrolMinX) { this.x = this.patrolMinX; this.vx = Math.abs(this.vx); }
    if (this.x + this.width >= this.patrolMaxX) { this.x = this.patrolMaxX - this.width; this.vx = -Math.abs(this.vx); }
    this.animTime += dt;
  }

  render(ctx) {
    if (!this.alive) return;
    const cx = this.x + this.width / 2;
    const headY = this.y + 5;
    const bodyTop = this.y + 10;
    const bodyBot = this.y + 20;
    const swing = Math.sin(this.animTime * 6) * 6;

    ctx.strokeStyle = '#c00';
    ctx.lineWidth = 2;
    ctx.lineCap = 'round';

    ctx.beginPath();
    ctx.arc(cx, headY, 5, 0, Math.PI * 2);
    ctx.stroke();

    ctx.beginPath();
    ctx.moveTo(cx, bodyTop);
    ctx.lineTo(cx, bodyBot);
    ctx.stroke();

    ctx.beginPath();
    ctx.moveTo(cx, bodyTop + 3);
    ctx.lineTo(cx - 8, bodyTop + 10 + swing);
    ctx.moveTo(cx, bodyTop + 3);
    ctx.lineTo(cx + 8, bodyTop + 10 - swing);
    ctx.stroke();

    ctx.beginPath();
    ctx.moveTo(cx, bodyBot);
    ctx.lineTo(cx - 5 + swing, bodyBot + 8);
    ctx.moveTo(cx, bodyBot);
    ctx.lineTo(cx + 5 - swing, bodyBot + 8);
    ctx.stroke();
  }
}

class Block {
  constructor(x, y, size = 32) {
    this.x = x;
    this.y = y;
    this.size = size;
    this.used = false;
  }

  render(ctx) {
    const s = this.size;
    if (!this.used) {
      ctx.fillStyle = '#daa520';
      ctx.fillRect(this.x, this.y, s, s);
      ctx.strokeStyle = '#b8860b';
      ctx.lineWidth = 2;
      ctx.strokeRect(this.x, this.y, s, s);
      ctx.fillStyle = '#fff';
      ctx.font = 'bold 18px sans-serif';
      ctx.textAlign = 'center';
      ctx.textBaseline = 'middle';
      ctx.fillText('?', this.x + s / 2, this.y + s / 2);
    } else {
      ctx.fillStyle = '#888';
      ctx.fillRect(this.x, this.y, s, s);
      ctx.strokeStyle = '#666';
      ctx.lineWidth = 2;
      ctx.strokeRect(this.x, this.y, s, s);
      ctx.fillStyle = '#aaa';
      ctx.font = 'bold 18px sans-serif';
      ctx.textAlign = 'center';
      ctx.textBaseline = 'middle';
      ctx.fillText('U', this.x + s / 2, this.y + s / 2);
    }
  }
}

class Level {
  constructor(data) {
    this.bg = data.bg || '#87ceeb';
    this.platforms = data.platforms || [];
    this.enemies = (data.enemies || []).map(e => new Enemy(e.x, e.y, e.patrolMinX, e.patrolMaxX));
    this.blocks = (data.blocks || []).map(b => new Block(b.x, b.y, b.size));
    this.gate = data.gate || null;
    this.exit = data.exit || null;
    this.spawn = data.spawn || { x: 50, y: 300 };
    this.gateUnlocked = false;
  }

  render(ctx) {
    ctx.fillStyle = this.bg;
    ctx.fillRect(0, 0, 800, 600);

    ctx.fillStyle = '#654321';
    ctx.strokeStyle = '#432';
    ctx.lineWidth = 2;
    for (const p of this.platforms) {
      const r = 4;
      ctx.beginPath();
      ctx.moveTo(p.x + r, p.y);
      ctx.lineTo(p.x + p.w - r, p.y);
      ctx.quadraticCurveTo(p.x + p.w, p.y, p.x + p.w, p.y + r);
      ctx.lineTo(p.x + p.w, p.y + p.h - r);
      ctx.quadraticCurveTo(p.x + p.w, p.y + p.h, p.x + p.w - r, p.y + p.h);
      ctx.lineTo(p.x + r, p.y + p.h);
      ctx.quadraticCurveTo(p.x, p.y + p.h, p.x, p.y + p.h - r);
      ctx.lineTo(p.x, p.y + r);
      ctx.quadraticCurveTo(p.x, p.y, p.x + r, p.y);
      ctx.closePath();
      ctx.fill();
      ctx.stroke();
    }

    for (const b of this.blocks) b.render(ctx);

    if (this.gate) {
      if (!this.gateUnlocked) {
        ctx.fillStyle = '#b22';
        ctx.fillRect(this.gate.x, this.gate.y, this.gate.w, this.gate.h);
        ctx.strokeStyle = '#800';
        ctx.lineWidth = 3;
        ctx.strokeRect(this.gate.x, this.gate.y, this.gate.w, this.gate.h);
        ctx.fillStyle = '#ff0';
        ctx.font = 'bold 14px sans-serif';
        ctx.textAlign = 'center';
        ctx.textBaseline = 'middle';
        ctx.fillText('🔒', this.gate.x + this.gate.w / 2, this.gate.y + this.gate.h / 2);
      } else {
        ctx.strokeStyle = '#0a0';
        ctx.lineWidth = 2;
        ctx.strokeRect(this.gate.x, this.gate.y, this.gate.w, this.gate.h);
      }
    }

    if (this.exit) {
      ctx.fillStyle = '#0f0';
      ctx.globalAlpha = 0.3;
      ctx.fillRect(this.exit.x, this.exit.y, this.exit.w, this.exit.h);
      ctx.globalAlpha = 1;
      ctx.strokeStyle = '#0a0';
      ctx.lineWidth = 2;
      ctx.strokeRect(this.exit.x, this.exit.y, this.exit.w, this.exit.h);
    }
  }
}

class ParticleBurst {
  constructor(x, y, color) {
    this.particles = [];
    this.life = 0.6;
    for (let i = 0; i < 12; i++) {
      const angle = (Math.PI * 2 * i) / 12 + Math.random() * 0.3;
      const speed = 80 + Math.random() * 120;
      this.particles.push({
        x, y,
        vx: Math.cos(angle) * speed,
        vy: Math.sin(angle) * speed - 60,
        size: 2 + Math.random() * 3,
        color
      });
    }
  }

  update(dt) {
    this.life -= dt;
    for (const p of this.particles) {
      p.x += p.vx * dt;
      p.y += p.vy * dt;
      p.vy += 400 * dt;
    }
    return this.life > 0;
  }

  render(ctx) {
    const alpha = Math.max(0, this.life / 0.6);
    ctx.globalAlpha = alpha;
    for (const p of this.particles) {
      ctx.fillStyle = p.color;
      ctx.fillRect(p.x - p.size / 2, p.y - p.size / 2, p.size, p.size);
    }
    ctx.globalAlpha = 1;
  }
}
