let currentLevelIndex = 0;
let level = null;
let player = null;
let enemies = [];
let blocks = [];
let particles = [];
let enemiesDefeated = 0;
let blocksUsed = 0;
let gatesUnlocked = 0;
let screen = 'title';
let lastTime = 0;

const keys = { left: false, right: false, jump: false, restart: false };
let jumpPressed = false;

const gameEl = document.getElementById('game');
const hudScore = document.getElementById('hud-score');
const hudLives = document.getElementById('hud-lives');
const hudLevel = document.getElementById('hud-level');
const hudAbility = document.getElementById('hud-ability');
const titleScreen = document.getElementById('title-screen');
const gameOverScreen = document.getElementById('game-over-screen');
const victoryScreen = document.getElementById('victory-screen');

function showScreen(name) {
  screen = name;
  titleScreen.style.display = name === 'title' ? 'flex' : 'none';
  gameOverScreen.style.display = name === 'gameover' ? 'flex' : 'none';
  victoryScreen.style.display = name === 'victory' ? 'flex' : 'none';
  gameEl.style.display = name === 'play' ? 'block' : 'none';
}

function startLevel(idx) {
  currentLevelIndex = idx;
  const data = LEVELS[idx];
  level = new Level(data);
  player = new Player(data.spawn.x, data.spawn.y);
  player.abilities = player.abilities || {};
  enemies = data.enemies.map(e => new Enemy(e.x, e.y, e.patrolLeft, e.patrolRight));
  blocks = data.blocks.map(b => new Block(b.x, b.y, b.type, b.reward));
  particles = [];
  jumpPressed = false;
  showScreen('play');
  updateHUD();
}

function respawn() {
  const data = LEVELS[currentLevelIndex];
  player.x = data.spawn.x;
  player.y = data.spawn.y;
  player.vx = 0;
  player.vy = 0;
  player.alive = true;
  player.invincible = 60;
}

function advanceLevel() {
  if (currentLevelIndex + 1 < LEVELS.length) {
    const savedAbilities = { ...player.abilities };
    const savedLives = player.lives;
    startLevel(currentLevelIndex + 1);
    player.abilities = savedAbilities;
    player.lives = savedLives;
  } else {
    showScreen('victory');
  }
}

function restartGame() {
  currentLevelIndex = 0;
  enemiesDefeated = 0;
  blocksUsed = 0;
  gatesUnlocked = 0;
  startLevel(0);
  player.lives = 3;
  player.abilities = {};
}

function updateHUD() {
  hudScore.textContent = 'Score: ' + enemiesDefeated;
  hudLives.textContent = 'Lives: ' + (player ? player.lives : 3);
  hudLevel.textContent = 'Level: ' + (currentLevelIndex + 1);
  hudAbility.textContent = player && player.abilities.doubleJump ? 'Double Jump ✓' : '';
}

function spawnParticles(x, y, color) {
  particles.push(new ParticleBurst(x, y, color));
}

function rectOverlap(a, b) {
  return a.x < b.x + b.w && a.x + a.w > b.x && a.y < b.y + b.h && a.y + a.h > b.y;
}

function update(dt) {
  if (screen !== 'play') return;

  // Player input
  player.vx = 0;
  if (keys.left) player.vx = -player.speed;
  if (keys.right) player.vx = player.speed;

  if (keys.jump && !jumpPressed) {
    player.jump();
    jumpPressed = true;
  }
  if (!keys.jump) jumpPressed = false;

  player.update(dt);

  // Enemies
  enemies.forEach(e => {
    if (e.alive) e.update(dt);
  });

  // Blocks
  blocks.forEach(b => b.update(dt));

  // Particles
  particles.forEach(p => p.update(dt));
  particles = particles.filter(p => p.alive);

  // Collision: player vs platforms
  const plats = level.platforms || [];
  plats.forEach(p => {
    const pb = { x: p.x, y: p.y, w: p.w, h: p.h };
    const pr = { x: player.x, y: player.y, w: player.w, h: player.h };
    if (rectOverlap(pr, pb)) {
      if (player.vy > 0 && player.y + player.h - player.vy * dt <= p.y + 4) {
        player.y = p.y - player.h;
        player.vy = 0;
        player.grounded = true;
        player.canDoubleJump = true;
      } else if (player.vy < 0 && player.y - player.vy * dt >= p.y + p.h - 4) {
        player.y = p.y + p.h;
        player.vy = 0;
      } else if (player.vx > 0) {
        player.x = p.x - player.w;
      } else if (player.vx < 0) {
        player.x = p.x + p.w;
      }
    }
  });

  // Collision: player vs enemies
  enemies.forEach(e => {
    if (!e.alive) return;
    const pr = { x: player.x, y: player.y, w: player.w, h: player.h };
    const er = { x: e.x, y: e.y, w: e.w, h: e.h };
    if (!rectOverlap(pr, er)) return;

    if (player.vy > 0 && player.y + player.h - 10 < e.y + e.h / 2) {
      e.alive = false;
      player.vy = -6;
      enemiesDefeated++;
      spawnParticles(e.x + e.w / 2, e.y, '#ff0');
      updateHUD();
    } else if (!player.invincible) {
      player.lives--;
      if (player.lives <= 0) {
        showScreen('gameover');
      } else {
        spawnParticles(player.x + player.w / 2, player.y + player.h / 2, '#f00');
        respawn();
      }
      updateHUD();
    }
  });

  // Collision: player vs blocks (bump from below)
  blocks.forEach(b => {
    if (b.used) return;
    const pr = { x: player.x, y: player.y, w: player.w, h: player.h };
    const br = { x: b.x, y: b.y, w: b.w, h: b.h };
    if (!rectOverlap(pr, br)) return;

    if (player.vy < 0 && player.y - player.vy * dt >= b.y + b.h - 4) {
      player.y = b.y + b.h;
      player.vy = 2;
      b.bump();
      blocksUsed++;
      if (b.reward === 'doubleJump') {
        player.abilities.doubleJump = true;
        spawnParticles(b.x + b.w / 2, b.y, '#0ff');
      } else {
        spawnParticles(b.x + b.w / 2, b.y, '#fa0');
      }
      updateHUD();
    }
  });

  // Collision: player vs gates
  const gates = level.gates || [];
  gates.forEach(g => {
    if (g.open) return;
    const gr = { x: g.x, y: g.y, w: g.w, h: g.h };
    const pr = { x: player.x, y: player.y, w: player.w, h: player.h };
    if (rectOverlap(pr, gr)) {
      if (g.requires === 'doubleJump' && player.abilities.doubleJump) {
        g.open = true;
        gatesUnlocked++;
        spawnParticles(g.x + g.w / 2, g.y + g.h / 2, '#0f0');
        updateHUD();
      } else {
        if (player.vx > 0) player.x = g.x - player.w;
        else if (player.vx < 0) player.x = g.x + g.w;
        else if (player.vy > 0) { player.y = g.y - player.h; player.vy = 0; player.grounded = true; }
        else if (player.vy < 0) { player.y = g.y + g.h; player.vy = 0; }
      }
    }
  });

  // Collision: player vs exit
  const exit = level.exit;
  if (exit) {
    const pr = { x: player.x, y: player.y, w: player.w, h: player.h };
    const er = { x: exit.x, y: exit.y, w: 30, h: 30 };
    if (rectOverlap(pr, er)) {
      advanceLevel();
    }
  }

  // Fall off screen
  if (player.y > 500) {
    player.lives--;
    if (player.lives <= 0) {
      showScreen('gameover');
    } else {
      respawn();
    }
    updateHUD();
  }

  // Invincibility timer
  if (player.invincible > 0) player.invincible--;
}

function render() {
  if (screen !== 'play') return;

  const ctx = gameEl.getContext ? gameEl.getContext('2d') : null;
  if (!ctx) return;

  const data = LEVELS[currentLevelIndex];
  ctx.fillStyle = data.background;
  ctx.fillRect(0, 0, 800, 450);

  // Platforms
  ctx.fillStyle = '#654321';
  const plats = level.platforms || [];
  plats.forEach(p => ctx.fillRect(p.x, p.y, p.w, p.h));

  // Gates
  const gates = level.gates || [];
  gates.forEach(g => {
    if (g.open) return;
    ctx.fillStyle = '#888';
    ctx.fillRect(g.x, g.y, g.w, g.h);
    ctx.fillStyle = '#aaa';
    ctx.fillRect(g.x + 4, g.y + 4, g.w - 8, g.h - 8);
  });

  // Blocks
  blocks.forEach(b => b.render(ctx));

  // Enemies
  enemies.forEach(e => { if (e.alive) e.render(ctx); });

  // Player
  if (player.invincible > 0 && Math.floor(player.invincible / 3) % 2 === 0) {
    ctx.globalAlpha = 0.4;
  }
  player.render(ctx);
  ctx.globalAlpha = 1;

  // Particles
  particles.forEach(p => p.render(ctx));

  // Exit marker
  if (level.exit) {
    ctx.fillStyle = '#0f0';
    ctx.fillRect(level.exit.x, level.exit.y, 30, 30);
    ctx.fillStyle = '#fff';
    ctx.font = '12px monospace';
    ctx.fillText('EXIT', level.exit.x + 2, level.exit.y + 20);
  }
}

function loop(timestamp) {
  requestAnimationFrame(loop);
  if (screen !== 'play') return;

  let dt = (timestamp - lastTime) / 1000;
  lastTime = timestamp;
  if (dt > 1 / 30) dt = 1 / 30;
  if (dt <= 0) dt = 1 / 60;

  update(dt);
  render();
}

// Input handling
window.addEventListener('keydown', e => {
  if (e.code === 'ArrowLeft' || e.code === 'KeyA') keys.left = true;
  if (e.code === 'ArrowRight' || e.code === 'KeyD') keys.right = true;
  if (e.code === 'Space' || e.code === 'ArrowUp' || e.code === 'KeyW') {
    keys.jump = true;
    if (screen === 'title') {
      restartGame();
    }
  }
  if (e.code === 'KeyR') {
    keys.restart = true;
    if (screen === 'gameover' || screen === 'victory') {
      restartGame();
    }
  }
  e.preventDefault();
});

window.addEventListener('keyup', e => {
  if (e.code === 'ArrowLeft' || e.code === 'KeyA') keys.left = false;
  if (e.code === 'ArrowRight' || e.code === 'KeyD') keys.right = false;
  if (e.code === 'Space' || e.code === 'ArrowUp' || e.code === 'KeyW') keys.jump = false;
  if (e.code === 'KeyR') keys.restart = false;
});
