// PlatformerMovement.ts
export class PlatformerMovement {
    private sprite: Phaser.Physics.Arcade.Sprite;
    private cursors: Phaser.Types.Input.Keyboard.CursorKeys;
    private config: MovementConfig;
    private coyoteTimer: number = 0;
    private jumpBufferTimer: number = 0;
    private hasDoubleJump: boolean = false;
    private canDoubleJump: boolean = false;
    
    constructor(sprite: Phaser.Physics.Arcade.Sprite, cursors: Phaser.Types.Input.Keyboard.CursorKeys, config: MovementConfig) {
        this.sprite = sprite;
        this.cursors = cursors;
        this.config = config;
    }
    
    update(delta: number) {
        // Horizontal movement
        if (this.cursors.left.isDown) {
            this.sprite.body!.velocity.x = -this.config.walkSpeed;
            this.sprite.setFlipX(true);
        } else if (this.cursors.right.isDown) {
            this.sprite.body!.velocity.x = this.config.walkSpeed;
            this.sprite.setFlipX(false);
        } else {
            this.sprite.body!.velocity.x = 0;
        }
        
        // Coyote time
        if (this.sprite.body!.blocked.down) {
            this.coyoteTimer = this.config.coyoteTime;
            this.canDoubleJump = this.hasDoubleJump;
        } else {
            this.coyoteTimer -= delta;
        }
        
        // Jump buffer
        const jumpPressed = Phaser.Input.Keyboard.JustDown(this.cursors.up) || Phaser.Input.Keyboard.JustDown(this.cursors.space);
        if (jumpPressed) {
            this.jumpBufferTimer = this.config.jumpBufferTime;
        } else {
            this.jumpBufferTimer -= delta;
        }
        
        // Jump
        if (this.jumpBufferTimer > 0 && this.coyoteTimer > 0) {
            this.sprite.body!.velocity.y = -this.config.jumpPower;
            this.jumpBufferTimer = 0;
            this.coyoteTimer = 0;
        }
        // Double jump
        else if (jumpPressed && this.canDoubleJump && !this.sprite.body!.blocked.down) {
            this.sprite.body!.velocity.y = -this.config.doubleJumpPower;
            this.canDoubleJump = false;
        }
    }
    
    enableDoubleJump() {
        this.hasDoubleJump = true;
        this.canDoubleJump = true;
    }
    
    get isDoubleJumpEnabled(): boolean {
        return this.hasDoubleJump;
    }
}

export class StateMachine {
    private states: Map<string, { enter?: () => void; update?: (dt: number) => void; exit?: () => void }>;
    private currentState: string = '';
    
    constructor() {
        this.states = new Map();
    }
    
    addState(name: string, config: { enter?: () => void; update?: (dt: number) => void; exit?: () => void }) {
        this.states.set(name, config);
    }
    
    setState(name: string) {
        if (this.currentState === name) return;
        const current = this.states.get(this.currentState);
        if (current?.exit) current.exit();
        this.currentState = name;
        const next = this.states.get(name);
        if (next?.enter) next.enter();
    }
    
    update(dt: number) {
        const current = this.states.get(this.currentState);
        if (current?.update) current.update(dt);
    }
    
    getState(): string {
        return this.currentState;
    }
}

    private jumpBufferTimer: number = 0;

            if (this.doubleJumpEnabled) {
                this.canDoubleJump = true;
            }
        } else {
            this.coyoteTimer -= delta;

        const jumpJustPressed = Phaser.Input.Keyboard.JustDown(this.jumpKey) || Phaser.Input.Keyboard.JustDown(this.spaceKey);
        
        if (jumpJustPressed) {
            this.jumpBufferTimer = this.config.jumpBufferTime;
        } else {
            this.jumpBufferTimer -= delta;

        if (this.jumpBufferTimer > 0 && this.coyoteTimer > 0) {

interface StateConfig {
    enter?: () => void;
    update?: (dt: number) => void;
    exit?: () => void;
}

export class StateMachine {
    private states: Map<string, StateConfig> = new Map();
    private currentStateKey: string = '';
    
    addState(name: string, config: StateConfig): this {
        this.states.set(name, config);
        return this;
    }
    
    setState(name: string): void {
        if (this.currentStateKey === name) return;
        const current = this.states.get(this.currentStateKey);
        if (current?.exit) current.exit();
        this.currentStateKey = name;
        const next = this.states.get(name);
        if (next?.enter) next.enter();
    }
    
    update(dt: number): void {
        const current = this.states.get(this.currentStateKey);
        if (current?.update) current.update(dt);
    }
    
    getCurrentState(): string {
        return this.currentStateKey;
    }
}

import Phaser from 'phaser';
import { Stickman } from '../characters/Stickman';
import { StickEnemy } from '../characters/StickEnemy';
import { GameState, updateSnapshot } from '../utils';
import { LEVEL_ORDER } from '../LevelManager';

export abstract class BaseLevelScene extends Phaser.Scene {
    protected player!: Stickman;
    protected platforms!: Phaser.Physics.Arcade.StaticGroup;
    protected enemies!: Phaser.Physics.Arcade.Group;
    protected questionBlocks!: Phaser.Physics.Arcade.StaticGroup;
    protected gates!: Phaser.Physics.Arcade.Group;
    protected exitGate!: Phaser.Physics.Arcade.Sprite;
    protected levelKey: string = '';
    
    protected abstract getWorldSize(): { width: number; height: number };
    protected abstract getBackgroundColor(): string;
    protected abstract getSpawnPoint(): { x: number; y: number };
    protected abstract createLevelContent(): void;
    protected abstract getLevelName(): string;
    
    create(): void {
        // Reset per-level state
        this.enemiesDefeated = new Set();
        this.blocksUsed = new Set();
        this.gatesUnlocked = new Set();
        
        const worldSize = this.getWorldSize();
        
        // Set up world
        this.physics.world.setBounds(0, 0, worldSize.width, worldSize.height);
        this.cameras.main.setBounds(0, 0, worldSize.width, worldSize.height);
        this.cameras.main.setBackgroundColor(this.getBackgroundColor());
        
        // Create groups
        this.platforms = this.physics.add.staticGroup();
        this.enemies = this.physics.add.group();
        this.questionBlocks = this.physics.add.staticGroup();
        this.gates = this.physics.add.group();
        
        // Create level content
        this.createLevelContent();
        
        // Create player
        const spawn = this.getSpawnPoint();
        this.player = new Stickman(this, spawn.x, spawn.y);
        
        // Restore double jump if already acquired
        if (GameState.hasDoubleJump) {
            this.player.grantDoubleJump();
        }
        
        // Set up collisions
        this.setupCollisions();
        
        // Camera follow
        this.cameras.main.startFollow(this.player, true, 0.1, 0.1);
        this.cameras.main.setDeadzone(100, 50);
        
        // Launch UI
        this.scene.launch('UIScene');
        
        updateSnapshot();
    }
    
    protected enemiesDefeated!: Set<string>;
    protected blocksUsed!: Set<string>;
    protected gatesUnlocked!: Set<string>;
    
    protected setupCollisions(): void {
        // Player vs platforms
        this.physics.add.collider(this.player, this.platforms);
        
        // Player vs enemies
        this.physics.add.collider(this.player, this.enemies, this.handleEnemyCollision as Phaser.Types.Physics.Arcade.ArcadePhysicsCallback, undefined, this);
        
        // Player vs question blocks
        this.physics.add.collider(this.player, this.questionBlocks, this.handleBlockCollision as Phaser.Types.Physics.Arcade.ArcadePhysicsCallback, undefined, this);
        
        // Enemies vs platforms
        this.physics.add.collider(this.enemies, this.platforms);
        
        // Player vs gates (collider for blocking)
        this.physics.add.collider(this.player, this.gates, this.handleGateCollision as Phaser.Types.Physics.Arcade.ArcadePhysicsCallback, undefined, this);
        
        // Player vs exit gate (overlap)
        if (this.exitGate) {
            this.physics.add.overlap(this.player, this.exitGate, this.handleExit, undefined, this);
        }
    }
    
    protected handleEnemyCollision(player: Phaser.Types.Physics.Arcade.GameObjectWithBody, enemy: Phaser.Types.Physics.Arcade.GameObjectWithBody): void {
        const p = player as Stickman;
        const e = enemy as StickEnemy;
        
        if (e.isDefeated()) return;
        
        const pBody = p.body as Phaser.Physics.Arcade.Body;
        const eBody = e.body as Phaser.Physics.Arcade.Body;
        
        // Check if player is stomping (coming from above)
        if (pBody.velocity.y > 0 && pBody.bottom <= eBody.top + 20) {
            e.defeat();
            pBody.velocity.y = -350; // Bounce
            GameState.score += 100;
            GameState.enemiesDefeated.push(e.getData('id') || 'enemy');
            updateSnapshot();
        } else {
            // Player takes damage
            this.playerTakeDamage();
        }
    }
    
    protected handleBlockCollision(player: Phaser.Types.Physics.Arcade.GameObjectWithBody, block: Phaser.Types.Physics.Arcade.GameObjectWithBody): void {
        const p = player as Stickman;
        const b = block as Phaser.Physics.Arcade.Sprite;
        
        if (b.getData('used')) return;
        
        const pBody = p.body as Phaser.Physics.Arcade.Body;
        
        // Check if player is bumping from below
        if (pBody.touching.up && b.getData('type') === 'question') {
            this.bumpBlock(b, p);
        }
    }
    
    protected bumpBlock(block: Phaser.Physics.Arcade.Sprite, player: Stickman): void {
        block.setData('used', true);
        block.setTexture('qblock_used');
        
        const blockId = block.getData('id') || 'block';
        GameState.blocksUsed.push(blockId);
        
        // Grant double jump if not already acquired
        if (!GameState.hasDoubleJump) {
            GameState.hasDoubleJump = true;
            player.grantDoubleJump();
            this.showAbilityNotification('Double Jump Acquired!');
        } else {
            GameState.score += 50;
        }
        
        // Bump animation
        this.tweens.add({
            targets: block,
            y: block.y - 10,
            duration: 100,
            yoyo: true
        });
        
        updateSnapshot();
    }
    
    protected handleGateCollision(player: Phaser.Types.Physics.Arcade.GameObjectWithBody, gate: Phaser.Types.Physics.Arcade.GameObjectWithBody): void {
        const g = gate as Phaser.Physics.Arcade.Sprite;
        
        if (g.getData('type') !== 'ability_gate') return;
        if (g.getData('opened')) return;
        
        if (GameState.hasDoubleJump) {
            g.setData('opened', true);
            g.setTexture('gate_open');
            (g.body as Phaser.Physics.Arcade.Body).enable = false;
            GameState.gatesUnlocked.push(g.getData('id') || 'gate');
            this.showAbilityNotification('Gate Opened!');
            updateSnapshot();
        }
    }
    
    protected handleExit(): void {
        const currentIndex = LEVEL_ORDER.indexOf(this.levelKey);
        
        if (currentIndex < LEVEL_ORDER.length - 1) {
            GameState.currentLevel = currentIndex + 2;
            this.scene.stop('UIScene');
            this.scene.start(LEVEL_ORDER[currentIndex + 1]);
        } else {
            this.scene.stop('UIScene');
            this.scene.start('VictoryScene');
        }
    }
    
    protected playerTakeDamage(): void {
        const isDead = this.player.takeDamage();
        
        if (isDead) {
            GameState.lives--;
            
            if (GameState.lives <= 0) {
                this.scene.stop('UIScene');
                this.scene.start('GameOverScene');
            } else {
                // Respawn
                this.scene.restart();
            }
        }
        
        updateSnapshot();
    }
    
    protected showAbilityNotification(text: string): void {
        const { width, height } = this.cameras.main;
        const notification = this.add.text(
            this.cameras.main.scrollX + width / 2,
            this.cameras.main.scrollY + height / 2 - 50,
            text,
            {
                fontSize: '28px',
                color: '#ffff00',
                fontFamily: 'Arial',
                stroke: '#000000',
                strokeThickness: 4
            }
        ).setOrigin(0.5).setDepth(100);
        
        this.tweens.add({
            targets: notification,
            y: notification.y - 60,
            alpha: 0,
            duration: 2000,
            onComplete: () => notification.destroy()
        });
    }
    
    update(time: number, delta: number): void {
        if (!this.player || !this.player.active) return;
        
        this.player.updateMovement(delta);
        
        // Check if player fell off world
        const worldSize = this.getWorldSize();
        if (this.player.y > worldSize.height + 50) {
            GameState.lives--;
            
            if (GameState.lives <= 0) {
                this.scene.stop('UIScene');
                this.scene.start('GameOverScene');
            } else {
                this.scene.restart();
            }
        }
        
        updateSnapshot();
    }
    
    // Helper methods for level creation
    protected createPlatform(x: number, y: number, width: number): Phaser.Physics.Arcade.Sprite {
        const platform = this.platforms.create(x, y, 'platform_tile') as Phaser.Physics.Arcade.Sprite;
        platform.setDisplaySize(width, 32);
        platform.refreshBody();
        return platform;
    }
    
    protected createEnemy(x: number, y: number, patrolLeft: number, patrolRight: number, id: string): StickEnemy {
        const enemy = new StickEnemy(this, x, y, patrolLeft, patrolRight);
        enemy.setData('id', id);
        this.enemies.add(enemy);
        return enemy;
    }
    
    protected createQuestionBlock(x: number, y: number, id: string): Phaser.Physics.Arcade.Sprite {
        const block = this.questionBlocks.create(x, y, 'qblock_full') as Phaser.Physics.Arcade.Sprite;
        block.setData('type', 'question');
        block.setData('used', false);
        block.setData('id', id);
        block.refreshBody();
        return block;
    }
    
    protected createAbilityGate(x: number, y: number, id: string): Phaser.Physics.Arcade.Sprite {
        const gate = this.gates.create(x, y, 'gate_locked') as Phaser.Physics.Arcade.Sprite;
        gate.setData('type', 'ability_gate');
        gate.setData('opened', false);
        gate.setData('id', id);
        gate.setImmovable(true);
        (gate.body as Phaser.Physics.Arcade.Body).setAllowGravity(false);
        return gate;
    }
    
    protected createExitGate(x: number, y: number): Phaser.Physics.Arcade.Sprite {
        this.exitGate = this.physics.add.sprite(x, y, 'gate_open');
        this.exitGate.setData('type', 'exit');
        (this.exitGate.body as Phaser.Physics.Arcade.Body).setAllowGravity(false);
        (this.exitGate.body as Phaser.Physics.Arcade.Body).setImmovable(true);
        return this.exitGate;
    }
}

            fontFamily: 'Arial'

            fontSize: '20px',
            color: '#ffff00',
            fontFamily: 'Arial'
        }).setOrigin(0.5);
        
        this.tweens.add({

            alpha: 0.3,

            yoyo: true,
            repeat: -1
        });
